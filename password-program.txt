# Password Lock Program

correct_password = "1234"

def lock_screen():
    print("🔐 Welcome to the Locked System")
    attempts = 3

    while attempts > 0:
        pwd = input("Enter password: ")
        if pwd == correct_password:
            print("✅ Access Granted!")
            return
        else:
            attempts -= 1
            print(f"❌ Incorrect! {attempts} attempt(s) left.\n")

    print("🚫 Too many wrong attempts. System locked.")

lock_screen()
